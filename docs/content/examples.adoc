== Random Blues

[source,crystal]
----
require "chitra/global_context"

size 600, 200

rect_s = 10
gap = 0
x = gap
y = gap
loop do
  if x > width
    x = gap
    y += rect_s + gap
  end

  break if y > height

  fill 0, 0, Random.new.rand(0.0...1.0)

  rect x, y, rect_s, rect_s
  x += rect_s + gap
end

save "random_blue_rects.png"
----

OR

[source,crystal]
----
require "chitra"

ctx = Chitra.new 600, 200

rect_s = 10
gap = 0
x = gap
y = gap
loop do
  if x > ctx.width
    x = gap
    y += rect_s + gap
  end

  break if y > ctx.height

  ctx.fill 0, 0, Random.new.rand(0.0...1.0)

  ctx.rect x, y, rect_s, rect_s
  x += rect_s + gap
end

ctx.save "random_blue_rects.png"
----

== Pattern

[source,crystal]
----
require "chitra/global_context"

size 600, 600
background 255/256, 88/256, 0

def pattern
  no_stroke

  x = 0
  y = 50
  (1..20).each do
    fill 255/256, 132/256, 0
    rect x, y, 150, 100
    rect x+50, y+100, 100, 150

    fill 255/256, 48/256, 0
    rect x, y+50, 150, 50
    rect x+100, y+100, 50, 150

    x += 150
    y += 150
  end
end

ty = -600.0
(0..4).each do
  translate 0, ty
  pattern
  translate 0, -ty
  ty += 300
end

save "./output/pattern.png"
----

    ctx.rect x, y+50, 150, 50
    ctx.rect x+100, y+100, 50, 150

    x += 150
    y += 150

end

ty = -600.0
(0..4).each do
  ctx.translate 0, ty
  pattern
  ctx.translate 0, -ty

ctx.save "./output/pattern.png"
